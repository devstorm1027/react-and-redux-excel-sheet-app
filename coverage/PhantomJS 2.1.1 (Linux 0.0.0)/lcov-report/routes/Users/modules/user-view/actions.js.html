export const sendInviteRequest = () => {
  return (dispatch, getState) => {
    dispatch(sendInviteSubmit())
    const user = selectors.getUser(getState())
    resendUserInvite(user.email_address)
      .then((res) => {
        dispatch(sendInviteComplete())
      })
      .catch(() => {})
  }
}
